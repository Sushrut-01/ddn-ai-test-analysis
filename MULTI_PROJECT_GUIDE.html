1. Project Selector Component

// src/components/ProjectSelector.jsx

import React, { useState, useEffect } from 'react';
import { getProjects } from '../services/api';

export default function ProjectSelector() {
    const [projects, setProjects] = useState([]);
    const [currentProject, setCurrentProject] = useState(null);

    useEffect(() => {
        loadProjects();
    }, []);

    const loadProjects = async () => {
        const response = await getProjects();
        setProjects(response.projects);

        // Restore last selected project
        const storedProjectId = localStorage.getItem('current_project_id');
        if (storedProjectId) {
            const project = response.projects.find(
                p => p.id === parseInt(storedProjectId)
            );
            if (project) setCurrentProject(project);
        }
    };

    const handleProjectChange = (projectId) => {
        const project = projects.find(p => p.id === parseInt(projectId));

        // Update localStorage
        localStorage.setItem('current_project_id', project.id);
        localStorage.setItem('current_project_slug', project.slug);

        // Reload page
        window.location.reload();
    };

    return (
        <div className="project-selector">
            <select onChange={(e) => handleProjectChange(e.target.value)}>
                {projects.map(project => (
                    <option key={project.id} value={project.id}>
                        {project.name} ({project.my_role})
                    </option>
                ))}
            </select>
        </div>
    );
}

// Helper to get current project ID
export const getCurrentProjectId = () => {
    const projectId = localStorage.getItem('current_project_id');
    if (!projectId) {
        throw new Error('No project selected');
    }
    return parseInt(projectId);
};

// Failures API (project-scoped)
export const failuresAPI = {
    getList: (params = {}) => {
        const projectId = getCurrentProjectId();
        return axios.get(
            `${API_BASE_URL}/api/projects/${projectId}/failures`,
            { params }
        );
    }
};

// Jira API (project-scoped)
export const jiraAPI = {
    createIssue: (data) => {
        const projectId = getCurrentProjectId();
        return axios.post(
            `${API_BASE_URL}/api/projects/${projectId}/jira/create-issue`,
            data
        );
    }
};

1. Project Access Middleware

# implementation/middleware/project_context.py

def require_project_access(required_role=None):
    """Decorator to validate project access"""
    def decorator(f):
        @wraps(f)
        def decorated_function(*args, **kwargs):
            # 1. Extract project_id from URL
            project_id = kwargs.get('project_id')

            # 2. Get user from JWT token
            token = request.headers.get('Authorization')
            payload = jwt.decode(token, JWT_SECRET)
            user_id = payload['user_id']

            # 3. Check if user has access to this project
            cur.execute("""
                SELECT role FROM user_projects
                WHERE user_id = %s AND project_id = %s
            """, (user_id, project_id))

            result = cur.fetchone()
            if not result:
                return jsonify({'error': 'Access denied'}), 403

            # 4. Store in request context
            g.project_id = project_id
            g.user_id = user_id
            g.project_role = result[0]

            return f(*args, **kwargs)
        return decorated_function
    return decorator

# implementation/dashboard_api_full.py

@app.route('/api/projects/<int:project_id>/failures', methods=['GET'])
@require_project_access(required_role='viewer')
def get_project_failures(project_id):
    """Get failures for specific project"""

    limit = int(request.args.get('limit', 50))

    # Query ONLY this project's failures
    cur.execute("""
        SELECT * FROM failure_analysis
        WHERE project_id = %s
        ORDER BY created_at DESC
        LIMIT %s
    """, (project_id, limit))

    failures = [dict(row) for row in cur.fetchall()]

    return jsonify({
        'failures': failures,
        'project_id': project_id
    })

@app.route('/api/projects/<int:project_id>/jira/create-issue', methods=['POST'])
@require_project_access(required_role='developer')
def create_jira_issue(project_id):
    """Create Jira issue in project's Jira"""

    # Get project-specific Jira config
    cur.execute("""
        SELECT jira_project_key, jira_url, jira_token
        FROM project_configurations
        WHERE project_id = %s
    """, (project_id,))

    config = cur.fetchone()
    jira_key = config['jira_project_key']  # e.g., 'GURU' for Guruttava

    # Create issue in project's Jira
    jira = JIRA(server=config['jira_url'], ...)
    new_issue = jira.create_issue({
        'project': {'key': jira_key},
        'summary': request.json['summary']
    })

    return jsonify({'jira_key': new_issue.key})
